function solution(numbers) {
    const answer = numbers.sort((a, b) => {
        const ab = Number(`${a}${b}`)
        const ba = Number(`${b}${a}`)

        return ba - ab
    })
    
    return answer[0] === 0 ? '0' : answer.join('')
}
