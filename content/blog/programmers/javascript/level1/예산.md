function solution(d, budget) {
    let answer = d.length;
    const sortingArr = d.sort((a, b) => a - b);
    
    for(let i = 0; i < sortingArr.length; i++){
        const temp = budget - sortingArr[i]
        
        if(temp >= 0){
            budget = temp
        } else {
            answer = i
            break;
        }
    }
    return answer
}
