```js
function solution(s) {
    return s.split('').sort((a, b) => b.charCodeAt() - a.charCodeAt()).join('');
}
```

```js
function solution(s) {
    return s.split('')
        .sort((a, b) => a < b ? 1 : a > b ? -1 : 0)
        .join('')
}
```
