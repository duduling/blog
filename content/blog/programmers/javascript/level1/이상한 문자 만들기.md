function solution(s) {
    let answer = '';
    
    s.split(' ').map(item => {
        answer += ' '
        
        for(let i = 0; i < item.length; i++){
            const word = item[i]
            
            answer += i%2 === 0 ? word.toUpperCase() : word.toLowerCase()
        }    
    })
    
    return answer.slice(1)
}
