# 문자열 내 마음대로 정렬하기

## Solution

```js
function solution(strings, n) {
  return strings.sort((a, b) =>
    a[n] === b[n] ? (a > b ? 1 : -1) : a[n] > b[n] ? 1 : -1
  );
}

// 신박한 함수 -> localeCompare
function solution(strings, n) {
  return strings.sort((a, b) =>
    a[n] === b[n] ? a.localeCompare(b) : a[n].localeCompare(b[n])
  );
}
```
