# [카카오 인턴] 키패드 누르기

## Solution

```js
function solution(numbers, hand) {
  const HAND_CONSTANT = {
    left: "L",
    right: "R",
  };

  const keypad = {
    1: [0, 1],
    2: [0, 2],
    3: [0, 3],
    4: [1, 1],
    5: [1, 2],
    6: [1, 3],
    7: [2, 1],
    8: [2, 2],
    9: [2, 3],
    "*": [3, 1],
    0: [3, 2],
    "#": [3, 3],
  };
  let cacheHand = {
    L: "*",
    R: "#",
  };

  return numbers
    .map((number) => {
      switch (number) {
        case 1:
        case 4:
        case 7: {
          cacheHand[HAND_CONSTANT.left] = number;
          return HAND_CONSTANT.left;
        }
        case 2:
        case 5:
        case 8:
        case 0:
        case "*":
        case "#": {
          const [preLeftX, preLeftY] = keypad[cacheHand[HAND_CONSTANT.left]];
          const [preRightX, preRightY] = keypad[cacheHand[HAND_CONSTANT.right]];
          const [curX, curY] = keypad[number];

          const leftDistance =
            Math.abs(preLeftX - curX) + Math.abs(preLeftY - curY);
          const rightDistance =
            Math.abs(preRightX - curX) + Math.abs(preRightY - curY);

          const result =
            leftDistance === rightDistance
              ? HAND_CONSTANT[hand]
              : leftDistance < rightDistance
              ? HAND_CONSTANT.left
              : HAND_CONSTANT.right;
          cacheHand[result] = number;

          return result;
        }
        case 3:
        case 6:
        case 9: {
          cacheHand[HAND_CONSTANT.right] = number;
          return HAND_CONSTANT.right;
        }
        default: {
          console.log("Error");
        }
      }
    })
    .join("");
}
```
